def find_mode (arr: list):
  moda_count = 0
  k = 0
  arr = sorted(arr)
  for i in range (len(arr)-1):
    if arr[i] == arr[i+1]:
      k += 1
    else:
      if k > moda_count:
        moda_count = k
        moda = arr[i]
      k = 0
  return moda
